296  /// \brief Fill the device with a specific value
297  void fill(const Scalar_T C) {
298  const size_t B = sqrt(device.get_max_workgroup_size());
299 
300  Q.submit([&](sycl::handler &h) {
301  const size_t global_size_rows = ((rows + B - 1) / B) * B;
302  const size_t global_size_cols = ((cols + B - 1) / B) * B;
303  const auto M = rows;
304  const auto N = cols;
305  const auto A = data_device;
306 
307  sycl::range<2> global{global_size_rows, global_size_cols};
308  sycl::range<2> local{B, B};
309 
310  h.parallel_for(sycl::nd_range<2>(global, local),
311  [=](sycl::nd_item<2> it) {
312  const auto i = it.get_global_id(0);
313  const auto j = it.get_global_id(1);
314 
315  if (i >= M || j >= N)
316  return;
317 
318  A[i * N + j] = C;
319  });
320  }).wait();
321  }

411  ///////////////////////////////////////////////////////////////////////
412  /// \brief Function to perform binary matrix operations
413  template <BinaryOperation op>
414  void binary_matrix_operations(const Matrix &B, Matrix &C) const {
415  const auto rows = this->num_rows();
416  const auto cols = this->num_cols();
417 
418  if (rows != B.num_rows() || cols != B.num_cols()) {
419  throw std::runtime_error("ERROR: Incompatible Matrix dimensions.");
420  }
421 
422  const auto platform_idx = this->device.get_platform_index();
423  const auto device_idx = this->device.get_device_index();
424 
425  if (platform_idx != B.device.get_platform_index() ||
426  device_idx != B.device.get_device_index()) {
427  throw std::runtime_error("ERROR: Incompatible PySYCL device.");
428  }
429 
430  const size_t wg_size = sqrt(this->device.get_max_workgroup_size());
431 
432  Q.submit([&](sycl::handler &h) {
433  const size_t global_size_rows =
434  ((rows + wg_size - 1) / wg_size) * wg_size;
435  const size_t global_size_cols =
436  ((cols + wg_size - 1) / wg_size) * wg_size;
437 
438  sycl::range<2> global{global_size_rows, global_size_cols};
439  sycl::range<2> local{wg_size, wg_size};
440 
441  auto data_1 = this->get_device_data_ptr();
442  auto data_2 = B.get_device_data_ptr();
443  auto data_new = C.get_device_data_ptr();
444 
445  h.parallel_for(sycl::nd_range<2>(global, local),
446  [=](sycl::nd_item<2> it) {
447  const auto i = it.get_global_id(0);
448  const auto j = it.get_global_id(1);
449 
450  if (i >= rows || j >= cols)
451  return;
452 
453  if constexpr (op == BinaryOperation::ADD) {
454  data_new[i * cols + j] =
455  data_1[i * cols + j] + data_2[i * cols + j];
456  } else if constexpr (op == BinaryOperation::SUBTRACT) {
457  data_new[i * cols + j] =
458  data_1[i * cols + j] - data_2[i * cols + j];
459  } else if constexpr (op == BinaryOperation::MULTIPLY) {
460  data_new[i * cols + j] =
461  data_1[i * cols + j] * data_2[i * cols + j];
462  } else if constexpr (op == BinaryOperation::DIVIDE) {
463  data_new[i * cols + j] =
464  data_1[i * cols + j] / data_2[i * cols + j];
465  }
466  });
467  }).wait();
468  }

470  ///////////////////////////////////////////////////////////////////////
471  /// \brief Function to perform reduction operations
472  template <typename Operation_T>
473  auto reductions(Operation_T op, Scalar_T val = 0.0) {
474  sycl::buffer<Scalar_T> buf{&val, 1};
475 
476  const size_t wg_size = device.get_max_workgroup_size();
477 
478  Q.submit([&](sycl::handler &h) {
479  const auto reduction_func =
480  sycl::reduction(buf, h, std::forward<Operation_T>(op));
481 
482  const size_t global_size =
483  ((rows * cols + wg_size - 1) / wg_size) * wg_size;
484  sycl::range<1> global{global_size};
485  sycl::range<1> local{wg_size};
486 
487  auto data_device_ptr = data_device;
488  const size_t N = rows * cols;
489 
490  h.parallel_for(sycl::nd_range<1>(global, local), reduction_func,
491  [=](sycl::nd_item<1> it, auto &el) {
492  const auto idx = it.get_global_id();
493  if (idx >= N)
494  return;
495 
496  el.combine(data_device_ptr[idx]);
497  });
498  }).wait();
499 
500  sycl::host_accessor val_host{buf, sycl::read_only};
501  val = val_host[0];
502 
503  return val;
504  }
