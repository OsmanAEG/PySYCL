31 /// \brief Matrix multiplication function
32 /// \param[in] A The first Array2D that is being multiplied.
33 /// \param[in] B The second Array2D that is being multiplied.
34 /// \return The Array2D product of the matrix multiplication.
35 Array2D matmul(Array2D& A, Array2D& B){
36  if(A.num_cols() != B.num_rows()) throw std::runtime_error("ERROR: Incompatible Array2D dimensions.");
37 
38  const bool same_platform_idx = A.get_device().get_platform_index() == B.get_device().get_platform_index();
39  const bool same_device_idx = A.get_device().get_device_index() == B.get_device().get_device_index();
40 
41  if(!same_platform_idx || !same_device_idx) throw std::runtime_error("ERROR: Incompatible PySYCL device.");
42 
43  const auto M = A.num_rows();
44  const auto N = A.num_cols();
45  const auto P = B.num_cols();
46 
47  const size_t wg_size = sqrt(A.get_device().get_max_workgroup_size());
48 
49  Array2D C(M, P, A.get_device());
50  auto Q = A.get_device().get_queue();
51 
52  Q.submit([&](sycl::handler& h){
53  const size_t global_size_M = ((M + wg_size - 1)/wg_size)*wg_size;
54  const size_t global_size_P = ((P + wg_size - 1)/wg_size)*wg_size;
55 
56  sycl::range<2> global{global_size_M, global_size_P};
57  sycl::range<2> local{wg_size, wg_size};
58 
59  auto data_1 = A.get_device_data_ptr();
60  auto data_2 = B.get_device_data_ptr();
61  auto data_new = C.get_device_data_ptr();
62 
63  h.parallel_for(sycl::nd_range<2>(global, local), [=](sycl::nd_item<2> it){
64  const auto i = it.get_global_id(0);
65  const auto j = it.get_global_id(1);
66 
67  if(i >= M || j >= P) return;
68 
69  float c_ij = 0.0;
70 
71  for(int n = 0; n < N; ++n){
72  c_ij += data_1[i*N + n]*data_2[n*P + j];
73  }
74 
75  data_new[i*P + j] = c_ij;
76  });
77  }).wait();
78 
79  return C;
80 }

115  auto data_2 = B.get_device_data_ptr();

122  const auto jG = it.get_global_id(1);
123

125  const auto jL = it.get_local_id(1);
126

141
