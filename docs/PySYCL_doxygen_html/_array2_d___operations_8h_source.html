39  const Array2D_Object &array2D_2,
40  const float &A = 1.0,
41  const float &B = 1.0){
42  // Check that the arrays are the same size.
43  if(array2D_1.get_rows() != array2D_2.get_rows() ||
44  array2D_1.get_columns() != array2D_2.get_columns()){
45  throw std::runtime_error("Arrays are not the same size.");
46  }
47 
48  if(array2D_1.get_device().queue() != array2D_2.get_device().queue()){
49  throw std::runtime_error("Arrays are not on the same device.");
50  }
51 
52  const auto M = array2D_1.get_rows();
53  const auto N = array2D_1.get_columns();
54  const auto device = array2D_1.get_device();
55 
56  auto Q = device.queue();
57 
58  // Create a new array to store the result.
59  Array2D_Object result(M, N, device);
60 
61  // Create a command group to perform the addition.
62  Q.submit([&](sycl::handler &h){
63  const auto data_1 = array2D_1.get_data_device();
64  const auto data_2 = array2D_2.get_data_device();
65  const auto data_3 = result.get_data_device();
66 
67  h.parallel_for(M*N, [=](sycl::id<1> idx){
