36 /// \brief Array2D-Array2D - element-wise operation.
37 /// \param[in] arr2D_1 First Array2D for operation.
38 /// \param[in] arr2D_2 Second Array2D for operation.
39 /// \param[in] op The operation to perform.
40 /// \param[in] A Constant multiplier for the first Array2D.
41 /// \param[in] B Constant multiplier for the second Array2D.
42 /// \return The result of the operation.
43 template <typename Array2D_type, typename Operation_type>
44 Array2D_type Array2D_Element_Wise_Operation(const Array2D_type &arr2D_1,
45  const Array2D_type &arr2D_2,
46  const Operation_type &op,
47  const float &A = 1.0f,
48  const float &B = 1.0f) {
49  // Check that the two arrays are the same size
50  if (arr2D_1.number_of_rows() != arr2D_2.number_of_rows() ||
51  arr2D_1.number_of_cols() != arr2D_2.number_of_cols()) {
52  throw std::runtime_error("Array2D objects must be the same size to add.");
53  }
54 
55  // Check that the two arrays are on the same device
56  if (arr2D_1.get_device().queue() != arr2D_2.get_device().queue()) {
57  throw std::runtime_error("Array2D objects must be on the same device to add.");
58  }
59 
60  const auto rows = arr2D_1.number_of_rows();
61  const auto cols = arr2D_1.number_of_cols();
62  const auto device = arr2D_1.get_device();
63 
64  auto Q = device.queue();
65 
66  // Create a new Array2D object to store the result
67  Array2D_type result(rows, cols, device);
68 
69  // Create a command group to perform the addition
70  Q.submit([&](sycl::handler &h){
71  const auto data_1 = arr2D_1.get_data_ptr();
72  const auto data_2 = arr2D_2.get_data_ptr();
73  const auto data_r = result.get_data_ptr();
74 
75  h.parallel_for(sycl::range<2>(rows, cols), [=](sycl::id<2> idx){
76  int i = idx[0];
77  int j = idx[1];
78 
79  data_r[i*cols + j] = op(A*data_1[i*cols + j], B*data_2[i*cols + j]);
80  });
81  });
82 
83  return result;
84 }
