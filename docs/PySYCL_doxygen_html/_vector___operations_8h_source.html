34///////////////////////////////////////////////////////////////////////
35/// \brief Function for element-wise vector operations in SYCL.
36/// This function receives two vectors and returns a single
37/// vector.
38/// \tparam Function_T Type of operation to be performed.
39/// \param vector_a First vector.
40/// \param vector_b Second vector.
41/// \param Q SYCL queue.
42/// \param function Operation to be performed.
43/// \return Vector resulting from the operation.
44template <typename Function_T>
45std::vector<double> Vector_Operation(std::vector<double> vector_a,
46 std::vector<double> vector_b,
47 sycl::queue Q,
48 Function_T function){
49 if(vector_a.size() != vector_b.size()){
50 throw std::runtime_error("Error: vectors must have the same size.");
51 }
52
53 const size_t N = vector_a.size();
54
55 std::vector<double> vector_c(vector_a.size());
56
57 {
58 sycl::buffer<double> buffer_a(vector_a);
59 sycl::buffer<double> buffer_b(vector_b);
60 sycl::buffer<double> buffer_c(vector_c);
61
62 Q.submit([&](sycl::handler& h){
63 sycl::accessor acc_a(buffer_a, h, sycl::read_only);
64 sycl::accessor acc_b(buffer_b, h, sycl::read_only);
65 sycl::accessor acc_c(buffer_c, h, sycl::write_only);
66
67 h.parallel_for(sycl::range<1>(N), [=](sycl::id<1> idx){
68 acc_c[idx] = function(acc_a[idx], acc_b[idx]);
69 });
70 }).wait();
71 }
72
73 return vector_c;
74};

128///////////////////////////////////////////////////////////////////////
129/// \brief Function for vector sum reduction.
130/// \param vector_a Input vector.
131/// \param platform_index Index of the sycl platform to select.
132/// \param device_index Index of the sycl device to select.
133double Vector_Sum_Reduction(std::vector<double> vector_a,
134 pysycl::Device_Object device){
135
136 const size_t N = vector_a.size();
137
138 double vector_sum = 0.0;
139
140 {
141 sycl::buffer<double> buffer_a(vector_a);
142 sycl::buffer<double> buffer_vector_sum(&vector_sum, 1);
143
144 device.get_queue().submit([&](sycl::handler& h){
145 sycl::accessor acc_a(buffer_a, h, sycl::read_only);
146
147 auto sum_reduction = sycl::reduction(buffer_vector_sum, h, sycl::plus<double>());
148
149 h.parallel_for(sycl::range<1>(N), sum_reduction, [=](sycl::id<1> idx, auto& sum){
150 sum.combine(acc_a[idx]);
151 });
152 }).wait();
153 }
154
155 return vector_sum;
156}
