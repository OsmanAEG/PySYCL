159 std::vector<std::vector<float>> data_host_2d(M, std::vector<float>(N));
160 for (size_t i = 0; i < M; ++i) {

168///////////////////////////////////////////////////////////////////////
169void Array2D_Object::fill(float value) {
170 Q.submit([&](sycl::handler &h){
171 const auto data_device_ptr = data_device;
172 h.parallel_for(M*N, [=](sycl::id<1> idx){
173 data_device_ptr[idx] = value;
174 });
175 }).wait();

183///////////////////////////////////////////////////////////////////////
184void Array2D_Object::fill_random_host(float min_val, float max_val) {
185 float range = max_val - min_val;
186 for (size_t i = 0; i < M; ++i) {
187 for (size_t j = 0; j < N; ++j) {
188 float random_value = static_cast<float>(rand()) / static_cast<float>(RAND_MAX);
189 data_host[i*N + j] = (random_value * range) + min_val;
190 }
191 }
192}

195///////////////////////////////////////////////////////////////////////
196float Array2D_Object::sum_reduction() {
197 float array_sum = 0.0;
198
199 {
200 sycl::buffer<float> buffer_array_sum(&array_sum, 1);
201 Q.submit([&](sycl::handler& h){
202 const auto data_device_ptr = data_device;
203 auto sum_reduce = sycl::reduction(buffer_array_sum, h, sycl::plus<float>());
204
205 h.parallel_for(sycl::range<1>(M*N), sum_reduce, [=](sycl::id<1> idx, auto& sum){
206 sum.combine(data_device_ptr[idx]);
207 });
208 }).wait();
209 }
210
211 return array_sum;
212}
