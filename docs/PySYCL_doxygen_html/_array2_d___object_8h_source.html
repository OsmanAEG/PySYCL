198 void Array2D_Object::fill_random_host(float min_val, float max_val) {
199  float range = max_val - min_val;
200  for (size_t i = 0; i < M; ++i) {
201  for (size_t j = 0; j < N; ++j) {
202  float random_value = static_cast<float>(rand()) / static_cast<float>(RAND_MAX);
203  data_host[i*N + j] = (random_value * range) + min_val;
204  }
205  }
