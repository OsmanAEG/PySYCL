261 void pysycl::Array2D::fill(const double C){
262  const size_t B = sqrt(device.get_max_workgroup_size());
263 
264  Q.submit([&](sycl::handler& h){
265  const size_t global_size_rows = ((rows + B - 1)/B)*B;
266  const size_t global_size_cols = ((cols + B - 1)/B)*B;
267  const auto M = rows;
268  const auto N = cols;
269  const auto A = data;
270 
271  sycl::range<2> global{global_size_rows, global_size_cols};
272  sycl::range<2> local{B, B};
273 
274  h.parallel_for(sycl::nd_range<2>(global, local), [=](sycl::nd_item<2> it){
275  const auto i = it.get_global_id(0);
276  const auto j = it.get_global_id(1);
277 
278  if(i >= M || j >= N) return;
279 
280  A[i*N + j] = C;
281  });
282  }).wait();
283 }

286 pysycl::Array2D pysycl::Array2D::binary_matrix_operations(Array2D& B,
287  binary_operations op,
288  bool edit_self){
289  const auto rows = this->num_rows();
290  const auto cols = this->num_cols();
291 
292  if(rows != B.num_rows() || cols != B.num_cols()){
293  throw std::runtime_error("ERROR: Incompatible Array2D dimensions.");
294  }
295 
296  const auto platform_idx = this->device.get_platform_index();
297  const auto device_idx = this->device.get_device_index();
298 
299  if(platform_idx != B.device.get_platform_index() || device_idx != B.device.get_device_index()){
300  throw std::runtime_error("ERROR: Incompatible PySYCL device.");
301  }
302 
303  const size_t wg_size = sqrt(this->device.get_max_workgroup_size());
304 
305  Array2D* C;
306 
307  if(!edit_self) C = new Array2D(rows, cols, this->device);
308 
309  Q.submit([&](sycl::handler& h){
310  const size_t global_size_rows = ((rows + wg_size - 1)/wg_size)*wg_size;
311  const size_t global_size_cols = ((cols + wg_size - 1)/wg_size)*wg_size;
312 
313  sycl::range<2> global{global_size_rows, global_size_cols};
314  sycl::range<2> local{wg_size, wg_size};
315 
316  auto data_1 = this->get_data_ptr();
317  auto data_2 = B.get_data_ptr();
318  auto data_new = edit_self ? data_1 : C->get_data_ptr();
319 
320  h.parallel_for(sycl::nd_range<2>(global, local), [=](sycl::nd_item<2> it){
321  const auto i = it.get_global_id(0);
322  const auto j = it.get_global_id(1);
323 
324  if(i >= rows || j >= cols) return;
325 
326  switch(op){
327  case binary_operations::ADD:
328  data_new[i*cols + j] = data_1[i*cols + j] + data_2[i*cols + j];
329  break;
330 
331  case binary_operations::SUBTRACT:
332  data_new[i*cols + j] = data_1[i*cols + j] - data_2[i*cols + j];
333  break;
334 
335  case binary_operations::MULTIPLY:
336  data_new[i*cols + j] = data_1[i*cols + j] * data_2[i*cols + j];
337  break;
338 
339  case binary_operations::DIVIDE:
340  data_new[i*cols + j] = data_1[i*cols + j] / data_2[i*cols + j];
341  break;
342  }
343  });
344  }).wait();
345 
346  if(edit_self){
347  return *this;
348  }else{
349  pysycl::Array2D result = *C;
350  delete C;
351  return result;
352  }
353 }
