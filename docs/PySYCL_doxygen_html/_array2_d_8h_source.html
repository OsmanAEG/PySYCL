293  const size_t B = sqrt(device.get_max_workgroup_size());

308 
309  if(i >= M || j >= N) return;

311  A[i*N + j] = C;
312  });

360  ///////////////////////////////////////////////////////////////////////
361  /// \brief Function to perform binary matrix operations
362  template<BinaryOperation op>
363  void binary_matrix_operations(const Array2D& B, Array2D& C) const {
364  const auto rows = this->num_rows();
365  const auto cols = this->num_cols();
366 
367  if(rows != B.num_rows() || cols != B.num_cols()){
368  throw std::runtime_error("ERROR: Incompatible Array2D dimensions.");
369  }
370 
371  const auto platform_idx = this->device.get_platform_index();
372  const auto device_idx = this->device.get_device_index();
373 
374  if(platform_idx != B.device.get_platform_index() || device_idx != B.device.get_device_index()){
375  throw std::runtime_error("ERROR: Incompatible PySYCL device.");
376  }
377 
378  const size_t wg_size = sqrt(this->device.get_max_workgroup_size());
379 
380  Q.submit([&](sycl::handler& h){
381  const size_t global_size_rows = ((rows + wg_size - 1)/wg_size)*wg_size;
382  const size_t global_size_cols = ((cols + wg_size - 1)/wg_size)*wg_size;
383 
384  sycl::range<2> global{global_size_rows, global_size_cols};
385  sycl::range<2> local{wg_size, wg_size};
386 
387  auto data_1 = this->get_device_data_ptr();
388  auto data_2 = B.get_device_data_ptr();
389  auto data_new = C.get_device_data_ptr();
390 
391  h.parallel_for(sycl::nd_range<2>(global, local), [=](sycl::nd_item<2> it){
392  const auto i = it.get_global_id(0);
393  const auto j = it.get_global_id(1);
394 
395  if(i >= rows || j >= cols) return;
396 
397  if constexpr (op == BinaryOperation::ADD){
398  data_new[i*cols + j] = data_1[i*cols + j] + data_2[i*cols + j];
399  }else if constexpr(op == BinaryOperation::SUBTRACT){
400  data_new[i*cols + j] = data_1[i*cols + j] - data_2[i*cols + j];
401  }else if constexpr(op == BinaryOperation::MULTIPLY){
402  data_new[i*cols + j] = data_1[i*cols + j] * data_2[i*cols + j];
403  }else if constexpr(op == BinaryOperation::DIVIDE){
404  data_new[i*cols + j] = data_1[i*cols + j] / data_2[i*cols + j];
405  }
406  });
407  }).wait();
408  }

410  ///////////////////////////////////////////////////////////////////////
411  /// \brief Function to perform reduction operations
412  template<typename Operation_T>
413  auto reductions(Operation_T op, Scalar_T val = 0.0){
414  sycl::buffer<Scalar_T> buf{&val, 1};
415 
416  const size_t wg_size = device.get_max_workgroup_size();
417 
418  Q.submit([&](sycl::handler& h){
419  const auto reduction_func = sycl::reduction(buf, h, std::forward<Operation_T>(op));
420 
421  const size_t global_size = ((rows*cols + wg_size - 1)/wg_size)*wg_size;
422  sycl::range<1> global{global_size};
423  sycl::range<1> local{wg_size};
424 
425  auto data_device_ptr = data_device;
426  const size_t N = rows*cols;
427 
428  h.parallel_for(sycl::nd_range<1>(global, local), reduction_func, [=](sycl::nd_item<1> it, auto& el){
429  const auto idx = it.get_global_id();
430  if(idx >= N) return;
431 
432  el.combine(data_device_ptr[idx]);
433  });
434  }).wait();
435 
436  sycl::host_accessor val_host{buf, sycl::read_only};
437  val = val_host[0];
438 
439  return val;
440  }
