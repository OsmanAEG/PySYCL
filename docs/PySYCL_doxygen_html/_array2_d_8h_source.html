322 void pysycl::Array2D::fill(const float C){
323  const size_t B = sqrt(device.get_max_workgroup_size());
324 
325  Q.submit([&](sycl::handler& h){
326  const size_t global_size_rows = ((rows + B - 1)/B)*B;
327  const size_t global_size_cols = ((cols + B - 1)/B)*B;
328  const auto M = rows;
329  const auto N = cols;
330  const auto A = data_device;
331 
332  sycl::range<2> global{global_size_rows, global_size_cols};
333  sycl::range<2> local{B, B};
334 
335  h.parallel_for(sycl::nd_range<2>(global, local), [=](sycl::nd_item<2> it){
336  const auto i = it.get_global_id(0);
337  const auto j = it.get_global_id(1);
338 
339  if(i >= M || j >= N) return;
340 
341  A[i*N + j] = C;
342  });
343  }).wait();
344 }

347 pysycl::Array2D pysycl::Array2D::binary_matrix_operations(Array2D& B,
348  binary_operations op,
349  bool edit_self){
350  const auto rows = this->num_rows();
351  const auto cols = this->num_cols();
352 
353  if(rows != B.num_rows() || cols != B.num_cols()){
354  throw std::runtime_error("ERROR: Incompatible Array2D dimensions.");
355  }
356 
357  const auto platform_idx = this->device.get_platform_index();
358  const auto device_idx = this->device.get_device_index();
359 
360  if(platform_idx != B.device.get_platform_index() || device_idx != B.device.get_device_index()){
361  throw std::runtime_error("ERROR: Incompatible PySYCL device.");
362  }
363 
364  const size_t wg_size = sqrt(this->device.get_max_workgroup_size());
365 
366  Array2D C = edit_self ? *this : Array2D(rows, cols, this->device);
367 
368  Q.submit([&](sycl::handler& h){
369  const size_t global_size_rows = ((rows + wg_size - 1)/wg_size)*wg_size;
370  const size_t global_size_cols = ((cols + wg_size - 1)/wg_size)*wg_size;
371 
372  sycl::range<2> global{global_size_rows, global_size_cols};
373  sycl::range<2> local{wg_size, wg_size};
374 
375  auto data_1 = this->get_device_data_ptr();
376  auto data_2 = B.get_device_data_ptr();
377  auto data_new = C.get_device_data_ptr();
378 
379  h.parallel_for(sycl::nd_range<2>(global, local), [=](sycl::nd_item<2> it){
380  const auto i = it.get_global_id(0);
381  const auto j = it.get_global_id(1);
382 
383  if(i >= rows || j >= cols) return;
384 
385  switch(op){
386  case binary_operations::ADD:
387  data_new[i*cols + j] = data_1[i*cols + j] + data_2[i*cols + j];
388  break;
389 
390  case binary_operations::SUBTRACT:
391  data_new[i*cols + j] = data_1[i*cols + j] - data_2[i*cols + j];
392  break;
393 
394  case binary_operations::MULTIPLY:
395  data_new[i*cols + j] = data_1[i*cols + j] * data_2[i*cols + j];
396  break;
397 
398  case binary_operations::DIVIDE:
399  data_new[i*cols + j] = data_1[i*cols + j] / data_2[i*cols + j];
400  break;
401  }
402  });
403  }).wait();
404 
405  return C;
406 }

408 ///////////////////////////////////////////////////////////////////////
409 /// \brief Matrix multiplication function
410 /// \param[in] A The first Array2D that is being multiplied.
411 /// \param[in] B The second Array2D that is being multiplied.
412 /// \return The Array2D product of the matrix multiplication.
413 void pysycl::Array2D::matmul(Array2D& A, Array2D& B){
414  if(A.num_cols() != B.num_rows()){
415  throw std::runtime_error("ERROR: Incompatible Array2D dimensions.");
416  }
417 
418  if(rows != A.num_rows() || cols != B.num_cols()){
419  throw std::runtime_error("ERROR: Incompatible Array2D dimensions.");
420  }
421 
422  const bool same_platform_idx = A.get_device().get_platform_index() == B.get_device().get_platform_index();
423  const bool same_device_idx = A.get_device().get_device_index() == B.get_device().get_device_index();
424 
425  if(!same_platform_idx || !same_device_idx) throw std::runtime_error("ERROR: Incompatible PySYCL device.");
426 
427  if(device.get_platform_index() != A.get_device().get_platform_index()){
428  throw std::runtime_error("ERROR: Incompatible PySYCL device.");
429  }
430 
431  if(device.get_device_index() != A.get_device().get_device_index()){
432  throw std::runtime_error("ERROR: Incompatible PySYCL device.");
433  }
434 
435  const auto M = A.num_rows();
436  const auto N = A.num_cols();
437  const auto P = B.num_cols();
438 
439  const size_t wg_size = sqrt(device.get_max_workgroup_size());
440 
441  Q.submit([&](sycl::handler& h){
442  const size_t global_size_M = ((M + wg_size - 1)/wg_size)*wg_size;
443  const size_t global_size_P = ((P + wg_size - 1)/wg_size)*wg_size;
444 
445  sycl::range<2> global{global_size_M, global_size_P};
446  sycl::range<2> local{wg_size, wg_size};
447 
448  auto data_1 = A.get_device_data_ptr();
449  auto data_2 = B.get_device_data_ptr();
450 
451  auto data_device_ptr = data_device;
452 
453  h.parallel_for(sycl::nd_range<2>(global, local), [=](sycl::nd_item<2> it){
454  const auto i = it.get_global_id(0);
455  const auto j = it.get_global_id(1);
456 
457  if(i >= M || j >= P) return;
458 
459  float c_ij = 0.0;
460 
461  for(int n = 0; n < N; ++n){
462  c_ij += data_1[i*N + n]*data_2[n*P + j];
463  }
464 
465  data_device_ptr[i*P + j] = c_ij;
466  });
467  }).wait();
468 }
