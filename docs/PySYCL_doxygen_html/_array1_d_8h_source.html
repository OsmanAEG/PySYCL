352  ///////////////////////////////////////////////////////////////////////
353  /// \brief Function to perform binary matrix operations
354  template <BinaryOperation op>
355  void binary_vector_operations(const Array1D &B, Array1D &C) const {
356  const auto rows = this->get_size();
357 
358  if (size != B.get_size()) {
359  throw std::runtime_error("ERROR: Incompatible Array1D dimensions.");
360  }
361 
362  const auto platform_idx = this->device.get_platform_index();
363  const auto device_idx = this->device.get_device_index();
364 
365  if (platform_idx != B.device.get_platform_index() ||
366  device_idx != B.device.get_device_index()) {
367  throw std::runtime_error("ERROR: Incompatible PySYCL device.");
368  }
369 
370  const size_t wg_size = this->device.get_max_workgroup_size();
371 
372  Q.submit([&](sycl::handler &h) {
373  const size_t global_size = ((size + wg_size - 1) / wg_size) * wg_size;
374  const auto N = size;
375 
376  sycl::range<1> global{global_size};
377  sycl::range<1> local{wg_size};
378 
379  auto data_1 = this->get_device_data_ptr();
380  auto data_2 = B.get_device_data_ptr();
381  auto data_new = C.get_device_data_ptr();
382 
383  h.parallel_for(sycl::nd_range<1>(global, local),
384  [=](sycl::nd_item<1> it) {
385  const auto i = it.get_global_id();
386 
387  if (i >= N)
388  return;
389 
390  if constexpr (op == BinaryOperation::ADD) {
391  data_new[i] = data_1[i] + data_2[i];
392  } else if constexpr (op == BinaryOperation::SUBTRACT) {
393  data_new[i] = data_1[i] - data_2[i];
394  } else if constexpr (op == BinaryOperation::MULTIPLY) {
395  data_new[i] = data_1[i] * data_2[i];
396  } else if constexpr (op == BinaryOperation::DIVIDE) {
397  data_new[i] = data_1[i] / data_2[i];
398  }
399  });
400  }).wait();
401  }

403  ///////////////////////////////////////////////////////////////////////
404  /// \brief Function to perform reduction operations
405  template <typename Operation_T>
406  auto reductions(Operation_T &&op, Scalar_T val = 0.0) {
407  sycl::buffer<Scalar_T> buf{&val, 1};
408 
409  const size_t wg_size = device.get_max_workgroup_size();
410 
411  Q.submit([&](sycl::handler &h) {
412  const auto reduction_func =
413  sycl::reduction(buf, h, std::forward<Operation_T>(op));
414 
415  const size_t global_size = ((size + wg_size - 1) / wg_size) * wg_size;
416  sycl::range<1> global{global_size};
417  sycl::range<1> local{wg_size};
418 
419  auto data_device_ptr = data_device;
420  const size_t N = size;
421 
422  h.parallel_for(sycl::nd_range<1>(global, local), reduction_func,
423  [=](sycl::nd_item<1> it, auto &el) {
424  const auto idx = it.get_global_id();
425  if (idx >= N)
426  return;
427 
428  el.combine(data_device_ptr[idx]);
429  });
430  }).wait();
431 
432  sycl::host_accessor val_host{buf, sycl::read_only};
433  val = val_host[0];
434 
435  return val;
436  }
