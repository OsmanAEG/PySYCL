358 pysycl::Array1D pysycl::Array1D::binary_vector_operations(Array1D& B,
359  binary_operations op,
360  bool edit_self){
361  const auto rows = this->get_size();
362 
363  if(size != B.get_size()){
364  throw std::runtime_error("ERROR: Incompatible Array1D dimensions.");
365  }
366 
367  const auto platform_idx = this->device.get_platform_index();
368  const auto device_idx = this->device.get_device_index();
369 
370  if(platform_idx != B.device.get_platform_index() || device_idx != B.device.get_device_index()){
371  throw std::runtime_error("ERROR: Incompatible PySYCL device.");
372  }
373 
374  const size_t wg_size = this->device.get_max_workgroup_size();
375 
376  Array1D C = edit_self ? *this : Array1D(size, this->device);
377 
378  Q.submit([&](sycl::handler& h){
379  const size_t global_size = ((size + wg_size - 1)/wg_size)*wg_size;
380  const auto N = size;
381 
382  sycl::range<1> global{global_size};
383  sycl::range<1> local{wg_size};
384 
385  auto data_1 = this->get_device_data_ptr();
386  auto data_2 = B.get_device_data_ptr();
387  auto data_new = C.get_device_data_ptr();
388 
389  h.parallel_for(sycl::nd_range<1>(global, local), [=](sycl::nd_item<1> it){
390  const auto i = it.get_global_id();
391 
392  if(i >= N) return;
393 
394  switch(op){
395  case binary_operations::ADD:
396  data_new[i] = data_1[i] + data_2[i];
397  break;
398 
399  case binary_operations::SUBTRACT:
400  data_new[i] = data_1[i] - data_2[i];
401  break;
402 
403  case binary_operations::MULTIPLY:
404  data_new[i] = data_1[i] * data_2[i];
405  break;
406 
407  case binary_operations::DIVIDE:
408  data_new[i] = data_1[i] / data_2[i];
409  break;
410  }
411  });
412  }).wait();
413 
414  return C;

418 template<typename Operation_T>
419 auto pysycl::Array1D::reductions(Operation_T op, float val){
420  sycl::buffer<float> buf{&val, 1};
421 
422  const size_t wg_size = device.get_max_workgroup_size();
423 
424  Q.submit([&](sycl::handler& h){
425  const auto reduction_func = sycl::reduction(buf, h, op);
426 
427  const size_t global_size = ((size + wg_size - 1)/wg_size)*wg_size;
428  sycl::range<1> global{global_size};
429  sycl::range<1> local{wg_size};
430 
431  auto data_device_ptr = data_device;
432  const size_t N = size;
433 
434  h.parallel_for(sycl::nd_range<1>(global, local), reduction_func, [=](sycl::nd_item<1> it, auto& el){
435  const auto idx = it.get_global_id();
436  if(idx >= N) return;
437 
438  el.combine(data_device_ptr[idx]);
439  });
440  }).wait();
441 
442  sycl::host_accessor val_host{buf, sycl::read_only};
443  val = val_host[0];
444 
445  return val;
446 }
